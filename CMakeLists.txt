cmake_minimum_required(VERSION 3.7)
project(uGame)
add_definitions(-D_OS_UNIX)
add_definitions(-D_DEBUG)
add_definitions(-D_LOG2FILE)

set(CMAKE_CXX_STANDARD 11)
find_package(OpenGL REQUIRED)
include_directories(include)
set(SOURCE_FILES src/main.cpp src/utils/Utils.cpp include/utils/Utils.h include/Constants.h src/utils/Logger.cpp include/utils/Logger.h src/core/Application.cpp include/core/Application.h src/core/AssetsManager.cpp include/core/AssetsManager.h src/core/PackageStream.cpp include/core/PackageStream.h src/utils/MD5.cpp include/utils/MD5.h src/core/StateManager.cpp include/core/StateManager.h include/core/GameState.h src/state/StateLoading.cpp include/state/StateLoading.h src/state/StateMenu.cpp include/state/StateMenu.h src/gui/Control.cpp include/gui/Control.h src/gui/ControlsContainer.cpp include/gui/ControlsContainer.h src/gui/Window.cpp include/gui/Window.h src/core/Lang.cpp include/core/Lang.h src/gui/Button.cpp include/gui/Button.h src/gui/UI.cpp include/gui/UI.h src/gui/Label.cpp include/gui/Label.h src/state/StateSignIn.cpp include/state/StateSignIn.h src/core/Config.cpp include/core/Config.h src/gui/InputBox.cpp include/gui/InputBox.h src/state/StatePlay.cpp include/state/StatePlay.h src/game/world/Chunk.cpp include/game/world/Chunk.h src/state/StateHeroList.cpp include/state/StateHeroList.h src/gui/ControlHeroList.cpp include/gui/ControlHeroList.h include/game/Tile.h include/game/TileObject.h src/game/TileBlock.cpp include/game/TileBlock.h src/game/world/TileObjectBlock.cpp include/game/TileObjectBlock.h)
add_executable(uGame ${SOURCE_FILES})
set(CMAKE_MODULE_PATH "${CMAKE_CURRENT_LIST_DIR}/cmake_modules")
find_package(SFML REQUIRED system window graphics network audio)
if (SFML_FOUND)
    include_directories(${SFML_INCLUDE_DIR})
    target_link_libraries(uGame ${SFML_LIBRARIES} ${OPENGL_LIBRARIES})
endif()
